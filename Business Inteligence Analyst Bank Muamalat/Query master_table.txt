CREATE TABLE master_table AS    
    SELECT
        "Date" as order_date,
        CategoryName as category_name,
        ProdName as product_name,
        Price as product_price,
        Quantity as order_qty,
        Quantity * Price as total_sales,
        CustomerEmail as cust_email,
        CustomerCity as cust_city
    FROM Orders
    JOIN Customers ON Customers.CustomerID = Orders.CustomerID
    JOIN Products ON Products.ProdNumber = Orders.ProdNumber
    JOIN ProductCategory ON ProductCategory.CategoryID = Products.Category
    ORDER BY order_date;
